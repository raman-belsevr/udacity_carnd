import cv2
import numpy as np
import sklearn
from sklearn.model_selection import train_test_split
from sklearn.utils import shuffle
from random import randint
from scipy import ndimage

class CustomDataGen:
    def __init__(self, image_dir, driving_log, config):
        self.config = config
        self.image_dir = image_dir
        self.driving_log = driving_log
    
    @staticmethod
    def build_samples(driving_log):
        samples = []
        with open(driving_log) as csvfile:
            reader = csv.reader(csvfile)
            first_line = True
            for line in reader:
                if first_line:
                    first_line = False
                    continue 
                samples.append(line)
        return samples
    
    @staticmethod
    def turning_angle(float_angle, threshold):
        if float_angle <= -threshold or float_angle >= threshold:
            return True
        else:
            return False
                
    def flip_image(self, image, angle):
        image_flipped = np.fliplr(image)
        return image_flipped, -angle 
    
    def shift_image(self,image):
        delta = randint(3, 10)
        if randint(0,1) == 0:
             sign = 1 
        else: 
            sign = -1 
        delta = sign * delta
        sequence = (0.0, delta, 0.0) 
        return ndimage.shift(image, sequence, mode="nearest")
    
    def gray_scale(image):
        gray_image = cv2.cvtColor(image, cv2.COLOR_RGB2GRAY)
        return np.expand_dims(gray_image, axis = 2)

    def crop_image(self, image, left_crop, right_crop, top_crop, bottom_crop):
        x, y, h = image.shape
        return image[top_crop: x - bottom_crop, left_crop: y- right_crop, :]

    def add_left_right_images(self, sample, float_angle):
        images = []
        angles = []
        if self.config["add_side_images"] == True:
            left_image_path = sample[1].split('/')[-1]
            left_angle = float_angle + 0.25
            images.append(left_image_path)
            angles.append(left_angle)

            right_image_path = sample[2].split('/')[-1]
            right_angle = float_angle - 0.25
            images.append(right_image_path)
            angles.append(right_angle)
        return images, angles      
    
    def initialize_sample(self, sample):
        center_image_path = sample[0].split('/')[-1]
        float_angle = float(sample[3])
        return center_image_path, float_angle
    
    @staticmethod
    def get_nshifts_from_angle(angle):
        rounded = round(angle, 1)
        if abs(rounded) < 0.5:
            return 0
        elif abs(rounded) >= 0.5:
            return 50
        
    def pre_process(self, image_path, float_angle, dry_run = False):
        images = []
        angles = []
            # read image in rgb
        image_path = os.path.join(self.image_dir, image_path)
        image = cv2.imread(image_path)
        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
        image = self.crop_image(image, 0, 0, 58, 25)
        #print("cropped image shape {}".format(image.shape))
        #display_image(image.squeeze(), float_angle)
        image = cv2.resize(image, (200, 66))
        images.append(image)
        angles.append(float_angle)
            
        # add images, only if it is a turning angle
        flip = False
        if CustomDataGen.turning_angle(float_angle, flipping_threshold) is True:
           # add flipped image
            f_image, f_angle = self.flip_image(image, float_angle)
            images.append(f_image)
            angles.append(f_angle)
            flip = True
                
                # add translated images 
        if CustomDataGen.turning_angle(float_angle, translation_threshold) is True:
            n_shifts = CustomDataGen.get_nshifts_from_angle(float_angle)
            for index in range(n_shifts):
                s_image = self.shift_image(image)
                images.append(s_image)
                angles.append(float_angle)
                if flip:
                    sf_image, sf_angle = self.flip_image(image, float_angle)
                    angles.append(sf_angle)
                    images.append(sf_image)
            
        return images, angles
    
    def pre_process_angle(self, float_angle):
        angles = []
        angles.append(float_angle)
        # add flipped images, only if angle > flipping_threshold
        flip = False
        if CustomDataGen.turning_angle(float_angle, flipping_threshold):
            # add flipped image
            angles.append(-1 * float_angle)
            flip = True
                
        if CustomDataGen.turning_angle(float_angle, translation_threshold):
            # add translated images, only if angle > translation_threshold
            n_shifts = CustomDataGen.get_nshifts_from_angle(float_angle)
            for index in range(n_shifts):
                angles.append(float_angle)
                if flip:
                    angles.append(-float_angle)
                    
        return angles           
        
    def generate_angle_dataset(self, samples):
        final_angles = []
        num_samples = len(samples)
        angle_counts = {}
        shuffle(samples)
        for sample in samples:
            image_path, angle = self.initialize_sample(sample)
            count = angle_counts.get(round(angle, 1), 0)
            if count > max_samples_per_angle:
                continue
            angle_counts[round(angle, 1)] = count + 1
            angles = self.pre_process_angle(angle)
            final_angles.extend(angles)
        return final_angles
    
    def generate(self, samples, batch_size = 32):
        num_samples = len(samples)
        batch_counter = 0
        angle_counts = {}
        while True:
            shuffle(samples)
            batch_counter = 0
            angle_counts = {}
            for offset in range(0, num_samples, batch_size):
                begin = offset
                end = begin + batch_size
                batch_samples = samples[begin: end]

                final_images = []
                final_angles = []
                for batch_sample in batch_samples:
                    image_path, angle = self.initialize_sample(batch_sample)
                    images, float_angles = self.pre_process(image_path, angle)
                    count = angle_counts.get(round(angle, 1), 0)
                    if count > max_samples_per_angle:
                        continue
                    angle_counts[round(angle, 1)] = count + 1
                    final_images.extend(images)
                    final_angles.extend(float_angles)
                    
                X_train = np.array(final_images)
                y_train = np.array(final_angles)
                yield sklearn.utils.shuffle(X_train, y_train)

## LeNet in Keras

from keras.layers.convolutional import Conv2D
from keras.layers.pooling import MaxPooling2D
from keras.layers.core import Flatten, Dense, Dropout, Lambda

def LeNet_for_Driving(input_shape):    
    model = Sequential()
    
    # Prep-processing
    # gray scaling
    # model.add(Lambda(lambda x: to_gray_scale(x), input_shape=input_shape))
    # normalization
    model.add(Lambda(lambda x: (x / 255.0) - 0.5, input_shape=input_shape))
    # cropping
    model.add(Cropping2D(cropping=((15, 10), (0, 0)), data_format=None))
    
    # Convolution 1
    model.add(Conv2D(6, (5, 5),
            strides=(1, 1),
            padding = 'valid',
            activation ='relu',
            input_shape = input_shape))
    model.add(MaxPooling2D(pool_size=(2, 2)))
    model.add(Dropout(0.2))
    
    # Convolution 2 Input = 14x14x6.. Output = 5x5x16
    model.add(Conv2D(16, (5, 5), strides=(1, 1), padding = 'valid', activation="relu"))
    model.add(MaxPooling2D(pool_size=(2, 2)))
    model.add(Dropout(0.2))
    
    # Flatten 
    model.add(Flatten())
    
    # Fully Connected
    model.add(Dense(120))
    model.add(Dropout(0.2))
    model.add(Activation('sigmoid'))
    
    # Final Output
    model.add(Dense(1))

    return model

## NVIDIA Architecture in Keras

def nvidia_for_driving(input_shape):
    
    model = Sequential()
   
    # Input 66x200x3 Output 36x200x3
    model.add(Lambda(lambda x:  (x / 255.0) - 0.5, input_shape = input_shape))

    #model.add(Cropping2D(cropping =((5, 5), (0, 0)), data_format = None))
    
    # Input 36x200x3 Output 16x98x24
    model.add(Conv2D(24, (5, 5), strides = (1,1), padding = 'valid', activation = 'elu'))
    model.add(MaxPooling2D(pool_size=(2, 2)))

    # Input 16x98x24 Output 6x47x36
    model.add(Conv2D(36, (5, 5), strides = (2,2), padding = 'valid', activation = 'elu'))
    #model.add(MaxPooling2D(pool_size=(2, 2)))

    model.add(Conv2D(48, (5, 5), strides = (2,2), padding = 'valid', activation = 'elu'))
    #model.add(MaxPooling2D(pool_size=(2, 2)))
    
    model.add(Conv2D(64, (3, 3), padding = 'valid', activation = 'elu'))
    model.add(Dropout(dropout_keep_prob))

    model.add(Conv2D(64, (3, 3), padding = 'valid', activation = 'elu'))
    model.add(Dropout(dropout_keep_prob))
    
    model.add(Flatten())
    model.add(Dropout(dropout_keep_prob))
    model.add(Dense(100))
    model.add(Dropout(dropout_keep_prob))
    model.add(Dense(50))
    model.add(Dense(1))
    model.summary()
    
    return model
